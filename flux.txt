class OmniFluxDataParser:
    """数据解析器"""

    @staticmethod
    def parse_response(data: bytes) -> Optional[SensorData]:
        """解析响应数据"""
        if len(data) < 38:
            return None

        try:
            sensor = SensorData()

            offset = 0

            if len(data) >= offset + 4:
                sensor.time_sec = struct.unpack('<I', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.distance = struct.unpack('<I', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.agtron = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.roc = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.t1 = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.ror1 = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.t2 = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 4:
                sensor.ror2 = struct.unpack('<f', data[offset:offset+4])[0]
                offset += 4
            if len(data) >= offset + 1:
                sensor.t1_valid = data[offset]
                offset += 1
            if len(data) >= offset + 1:
                sensor.t2_valid = data[offset]
                offset += 1
            if len(data) >= offset + 2:
                sensor.boom1_count = struct.unpack('<H', data[offset:offset+2])[0]
                offset += 2
            if len(data) >= offset + 2:
                sensor.boom2_count = struct.unpack('<H', data[offset:offset+2])[0]

            return sensor
        except Exception:
            return None


class BluetoothDebugger:
    """蓝牙"""

    CHAR_UUID_PLAIN = "0000aa01-0000-1000-8000-00805f9b34fb"
    CHAR_UUID_ENCRYPT = "0000ff01-0000-1000-8000-00805f9b34fb"
    REQUEST_DATA_CMD = bytes([0xdf, 0xdf, 0x03, 0x05, 0x00, 0xc6])

    def __init__(self):
        self.client = None
        self._connected = False
        self._notify_enabled = False
        self._recv_queue = queue.Queue()
        self._running = False
        self._loop_thread = None
        self._loop = None
        self.write_char = None
        self.notify_char = None
        self.device_name = None
        self.device_address = None
        self.on_data_received = None
        self._silent = True

        self._auto_send_running = False
        self._auto_send_thread = None
        self._auto_send_interval = 1

    def _run_async(self, coro, timeout=30):
        import asyncio
        if not self._loop or not self._loop.is_running():
            self.start()
        future = asyncio.run_coroutine_threadsafe(coro, self._loop)
        return future.result(timeout=timeout)

    def start(self):
        import asyncio
        if self._loop_thread and self._loop_thread.is_alive():
            return
        self._running = True

        def run_loop():
            import asyncio
            self._loop = asyncio.new_event_loop()
            asyncio.set_event_loop(self._loop)
            self._loop.run_forever()

        self._loop_thread = threading.Thread(target=run_loop, daemon=True)
        self._loop_thread.start()
        for _ in range(20):
            if self._loop and self._loop.is_running():
                break
            time.sleep(0.05)

    def stop(self):
        self._running = False
        self.stop_auto_send()
        if self._loop and self._loop.is_running():
            self._loop.call_soon_threadsafe(self._loop.stop)

    def scan_omniflux_devices(self, timeout: float = 5.0) -> List[dict]:
        import asyncio
        from bleak import BleakScanner

        devices = []

        def detection_callback(device, advertisement_data):
            if device.name and "OmniFlux" in device.name:
                if not any(d['address'] == device.address for d in devices):
                    devices.append({
                        'name': device.name,
                        'address': device.address,
                        'rssi': advertisement_data.rssi if advertisement_data else -100
                    })

        async def scan():
            scanner = BleakScanner(detection_callback)
            await scanner.start()
            await asyncio.sleep(timeout)
            await scanner.stop()
            return devices

        result = self._run_async(scan(), timeout=timeout+1)
        return sorted(result, key=lambda x: x['rssi'], reverse=True)

    def connect(self, address: str) -> bool:
        import asyncio
        from bleak import BleakClient

        async def connect_async():
            try:
                # 超时时间
                self.client = BleakClient(address, timeout=15.0)
                await self.client.connect()

                if self.client.is_connected:
                    self._connected = True
                    self.device_address = address
                    try:
                        self.device_name = await self.client.get_device_name()
                    except:
                        self.device_name = address

                    await self._find_characteristics()

                    if self.notify_char:
                        await self.client.start_notify(
                            self.notify_char.uuid,
                            self._notification_handler
                        )
                        self._notify_enabled = True

                    self.start_auto_send()
                    return True
            except Exception:
                pass
            return False

        return self._run_async(connect_async(), timeout=20)

    async def _find_characteristics(self):
        self.write_char = None
        self.notify_char = None

        for service in self.client.services:
            for char in service.characteristics:
                char_uuid = char.uuid.lower()
                if char_uuid == self.CHAR_UUID_PLAIN.lower():
                    self.write_char = char
                    self.notify_char = char
                    self.current_channel = "plain"
                elif char_uuid == self.CHAR_UUID_ENCRYPT.lower() and not self.write_char:
                    self.write_char = char
                    self.notify_char = char
                    self.current_channel = "encrypt"

    def _notification_handler(self, sender: int, data: bytes):
        self._recv_queue.put(data)
        if self.on_data_received:
            self.on_data_received(data)

    def disconnect(self):
        self.stop_auto_send()

        async def disconnect_async():
            if self.client and self.client.is_connected:
                try:
                    if self.notify_char and self._notify_enabled:
                        await self.client.stop_notify(self.notify_char.uuid)
                except:
                    pass
                try:
                    await self.client.disconnect()
                except:
                    pass
            # 无论是否成功，重置状态
            self._connected = False
            self._notify_enabled = False
            self.client = None  # 清空引用

        try:
            self._run_async(disconnect_async(), timeout=5)
        except:
            # 即使异常也重置状态
            self._connected = False
            self._notify_enabled = False
            self.client = None

    def send(self, data: bytes) -> bool:
        if not self._connected or not self.client or not self.client.is_connected:
            return False
        if not self.write_char:
            return False

        async def send_async():
            await self.client.write_gatt_char(self.write_char.uuid, data)
            return True

        try:
            return self._run_async(send_async(), timeout=5)
        except:
            return False

    def send_hex(self, hex_str: str) -> bool:
        try:
            hex_str = hex_str.strip().replace(' ', '').replace('-', '').replace(':', '')
            if len(hex_str) % 2 != 0:
                hex_str = '0' + hex_str
            data = bytes.fromhex(hex_str)
            return self.send(data)
        except:
            return False

    def start_auto_send(self):
        if self._auto_send_running:
            return

        self._auto_send_running = True
        self._auto_send_thread = threading.Thread(target=self._auto_send_loop, daemon=True)
        self._auto_send_thread.start()

    def stop_auto_send(self):
        self._auto_send_running = False
        if self._auto_send_thread and self._auto_send_thread.is_alive():
            self._auto_send_thread.join(timeout=1)

    def _auto_send_loop(self):
        while self._auto_send_running and self._connected:
            if self.write_char:
                try:
                    self.send(self.REQUEST_DATA_CMD)
                except:
                    pass
            time.sleep(self._auto_send_interval)

    def set_auto_send_interval(self, interval: float):
        self._auto_send_interval = interval

    @property
    def is_connected(self) -> bool:
        return self._connected

    @property
    def is_notify_enabled(self) -> bool:
        return self._notify_enabled

    CHAR_UUID_PLAIN = "0000aa01-0000-1000-8000-00805f9b34fb"
    CHAR_UUID_ENCRYPT = "0000ff01-0000-1000-8000-00805f9b34fb"
    REQUEST_DATA_CMD = bytes([0xdf, 0xdf, 0x03, 0x05, 0x00, 0xc6])

    def __init__(self):
        self.client = None
        self._connected = False
        self._notify_enabled = False
        self._recv_queue = queue.Queue()
        self._running = False
        self._loop_thread = None
        self._loop = None
        self.write_char = None
        self.notify_char = None
        self.device_name = None
        self.device_address = None
        self.on_data_received = None
        self._silent = True

        self._auto_send_running = False
        self._auto_send_thread = None
        self._auto_send_interval = 1

    def _run_async(self, coro, timeout=30):
        import asyncio
        if not self._loop or not self._loop.is_running():
            self.start()
        future = asyncio.run_coroutine_threadsafe(coro, self._loop)
        return future.result(timeout=timeout)

    def start(self):
        import asyncio
        if self._loop_thread and self._loop_thread.is_alive():
            返回
        self._running = True

        def run_loop():
            import asyncio
            self._loop = asyncio.new_event_loop()
            asyncio.set_event_loop(self._loop)
            self._loop.run_forever()

        self._loop_thread = threading.Thread(target=run_loop, daemon=True)
        self._loop_thread.start()
        for _ in range(20):
            if self._loop and self._loop.is_running():
                break
            time.sleep(0.05)

    def stop(self):
        self._running = False
        self.stop_auto_send()
        if self._loop and self._loop.is_running():
            self._loop.call_soon_threadsafe(self._loop.stop)

    def scan_omniflux_devices(self, timeout: float = 5.0) -> List[dict]:
        import asyncio
        from bleak import BleakScanner

        devices = []

        def detection_callback(device, advertisement_data):
            if device.name and "OmniFlux" in device.name:
                if not any(d['address'] == device.address for d in devices):
                    devices.append({
                        'name': device.name,
                        'address': device.address,
                        'rssi': advertisement_data.rssi if advertisement_data else -100
                    })

        async def scan():
            scanner = BleakScanner(detection_callback)
            await scanner.start()
            await asyncio.sleep(timeout)
            await scanner.stop()
            返回设备

        result = self._run_async(scan(), timeout=timeout+1)
        return sorted(result, key=lambda x: x['rssi'], reverse=True)

    def connect(self, address: str) -> bool:
        import asyncio
        from bleak import BleakClient

        async def connect_async():
            try:
                # 超时时间
                self.client = BleakClient(address, timeout=15.0)
                await self.client.connect()

                if self.client.is_connected:
                    self._connected = True
                    self.device_address = address
                    try:
                        self.device_name = await self.client.get_device_name()
                    except:
                        self.device_name = address

                    await self._find_characteristics()

                    if self.notify_char:
                        await self.client.start_notify(
                            self.notify_char.uuid,
                            self._notification_handler
                        )
                        self._notify_enabled = True

                    self.start_auto_send()
                    返回 True
            except Exception:
                pass
            返回 False

        return self._run_async(connect_async(), timeout=20)

    async def _find_characteristics(self):
        self.write_char = None
        self.notify_char = None

        for service in self.client.services:
            for char in service.characteristics:
                char_uuid = char.uuid.lower()
                if char_uuid == self.CHAR_UUID_PLAIN.lower():
                    self.write_char = char
                    self.notify_char = char
                    self.current_channel = "plain"
                elif char_uuid == self.CHAR_UUID_ENCRYPT.lower() and not self.write_char:
                    self.write_char = char
                    self.notify_char = char
                    self.current_channel = "encrypt"

    def _notification_handler(self, sender: int, data: bytes):
        self._recv_queue.put(data)
        if self.on_data_received:
            self.on_data_received(data)

    def disconnect(self):
        self.stop_auto_send()

        async def disconnect_async():
            if self.client and self.client.is_connected:
                try:
                    if self.notify_char and self._notify_enabled:
                        await self.client.stop_notify(self.notify_char.uuid)
                除外：
                    跳过
                try:
                    await self.client.disconnect()
                except:
                    跳过
            # 无论是否成功，重置状态
            self._connected = False
            self._notify_enabled = False
            self.client = None  # 清空引用

        try:
            self._run_async(disconnect_async(), timeout=5)
        除外：
            # 即使异常也重置状态
            self._connected = False
            self._notify_enabled = False
            self.client = None

    def send(self, data: bytes) -> bool:
        if not self._connected or not self.client or not self.client.is_connected:
            返回 False
        if not self.write_char:
            返回 False

        async def send_async():
            await self.client.write_gatt_char(self.write_char.uuid, data)
            返回 True

        try:
            return self._run_async(send_async(), timeout=5)
        异常处理：
            返回 False

    def send_hex(self, hex_str: str) -> bool:
        try:
            hex_str = hex_str.strip().replace(' ', '').replace('-', '').replace(':', '')
            if len(hex_str) % 2!= 0:
                hex_str = '0' + hex_str
            data = bytes.fromhex(hex_str)
            return self.send(data)
        异常处理：
            返回 False

    def start_auto_send(self):
        if self._auto_send_running:
            返回

        self._auto_send_running = True
        self._auto_send_thread = threading.Thread(target=self._auto_send_loop, daemon=True)
        self._auto_send_thread.start()

    def stop_auto_send(self):
        self._auto_send_running = False
        if self._auto_send_thread and self._auto_send_thread.is_alive():
            self._auto_send_thread.join(timeout=1)

    def _auto_send_loop(self):
        while self._auto_send_running and self._connected:
            if self.write_char:
                try:
                    self.send(self.REQUEST_DATA_CMD)
                except:
                    跳过
            time.sleep(self._auto_send_interval)

    def set_auto_send_interval(self, interval: float):
        self._auto_send_interval = interval

    @属性
    def is_connected(self) -> bool:
        return self._connected

    @属性
    def is_notify_enabled(self) -> bool:
        返回 self._notify_enabled
